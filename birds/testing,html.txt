/* Clear Input Text */
function clearForm() {	
  var childNodes = document.querySelectorAll('.market-form input');
  for (var i=0; i<childNodes.length; i++) {   
    if (childNodes[i].type == "text") {
		childNodes[i].value = "";
	}	
  }
}
